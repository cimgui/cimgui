file(GLOB CIMGUI_TEST_SOURCES main.c)

add_executable(cimgui_test ${CIMGUI_TEST_SOURCES})

                                             ${CMAKE_BINARY_DIR})

                           PRIVATE CIMGUI_DEFINE_ENUMS_AND_STRUCTS=1)
